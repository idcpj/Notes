## dmeo
```
加载 progress.min.js  与 progressjs.min.css

//想要出现进度条时候
progressJs().start();
//也可以链式调用
progressJs().start().set(20);

//加载进度条
progressJs().increase(5)  //按百分之5加载

//结束加载
progressJs().end()
```
